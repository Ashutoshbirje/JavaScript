# Syntex
// Function Declaration (hoisting)
--> function name_fnx(para_){
     // function body
    }
// call
--> name_fnx();

// Named Function Assignment (No hoisting)
--> (let/var) var_name = function name_fnx(para_){
     // function body
    }
// call
--> var_name();

// Expression Function Assignment (No hoisting)
// Anonymous Function Assignment (No hoisting)
--> (let/var) var_name = function (para_){
     // function body
    }
// call
--> var_name();

// Arrow Function Assignment (No hoisting)
--> (let/var) var_name = (para_)=>{
     // function body
    }
// call
--> var_name();
